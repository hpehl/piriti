16       * 
17       * @param value
18       *            The string to be converted
19       * @param format
20       *            Must be a valid number format or {@code null}
21       * @return {@code null} if the value is {@code null}, empty or in the wrong
22       *         format, otherwise the converted integer
23       * @see name.pehl.gwt.piriti.client.converter.Converter#convert(java.lang.String,
24       *      java.lang.String)
25       */
26      @Override
27      public Integer convert(String value, String format)
28      {
29          if (isValid(value))
30          {
31              try
32              {
33                  if (format == null)
34                  {
35                      return Integer.valueOf(value);
36                  }
37                  else
38                  {
39                      return parseInteger(value, format);
40                  }
41              }
42              catch (NumberFormatException e)
43              {
44                  return null;
45              }
46          }
47          return null;
48      }

51      /**
52       * Parsing happens in an extra method so it can be overwritten in unit
53       * tests.
54       * 
55       * @param value
56       * @param format
57       * @return
58       */
59      protected Integer parseInteger(String value, String format)
60      {
61          NumberFormat numberFormat = NumberFormat.getFormat(format);
62          return new Integer((int) numberFormat.parse(value));
63      }
