104     /**
105      * Looks up a field handler based on the information provided in the field
106      * context. The lookup logic is implemented like this:
107      * <ol>
108      * <li>If the fields type is a primitive return the
109      * {@link DefaultFieldHandler}
110      * <li>if the fields type is an enum return {@link EnumFieldHandler}
111      * <li>If the fields type is an array return {@link ArrayFieldHandler}
112      * <li>Try to lookup the field handler by the fields type classname (this
113      * will resolve all types registered in
114      * {@link #registerInitialFieldHandlers()}
115      * <li>If no field handler return {@link XmlRegistryFieldHandler}.
116      * </ol>
117      * 
118      * @param fieldContext
119      * @return
120      */
121     public FieldHandler findFieldHandler(FieldContext fieldContext)
122     {
123         FieldHandler handler = null;
124         if (fieldContext.isPrimitive())
125         {
126             handler = new DefaultFieldHandler();
127         }
128         else if (fieldContext.isEnum())
129         {
130             handler = new EnumFieldHandler();
131         }
132         else if (fieldContext.isArray())
133         {
134             handler = new ArrayFieldHandler();
135         }
136         else
137         {
138             // Ask the registry for all other stuff (basic types,
139             // collections, maps, ...)
140             handler = registry.get(fieldContext.getFieldType().getQualifiedSourceName());
141             if (handler == null)
142             {
143                 // Delegate to the XmlRegistry to resolve other mapped models
144                 handler = new XmlRegistryFieldHandler();
145             }
146         }
147         return handler;
148     }
