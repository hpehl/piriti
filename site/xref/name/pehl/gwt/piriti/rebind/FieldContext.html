13   * Class which contains information needed to generate code for the XPath
14   * evaluation, conversion and assignment of a field. An instance of this class
15   * is generated in {@link XmlReaderCreator} and passed to the

39      public FieldContext(TypeOracle typeOracle, FieldHandlerRegistry handlerRegistry, JClassType modelType,
40              JType fieldType, String fieldName, String xpath, String format, String xmlVariable, String valueVariable)
41              throws UnableToCompleteException
42      {
43          this.typeOracle = typeOracle;
44          this.handlerRegistry = handlerRegistry;
45          this.modelType = modelType;
46  
47          JPrimitiveType primitiveType = fieldType.isPrimitive();
48          if (primitiveType != null) // isPrimitive() is not available here!
49          {
50              try
51              {
52                  // Use the boxed type for primitives
53                  this.fieldType = typeOracle.getType(primitiveType.getQualifiedBoxedSourceName());
54              }
55              catch (NotFoundException e)
56              {
57                  throw new UnableToCompleteException();
58              }
59          }
60          else
61          {
62              this.fieldType = fieldType;
63          }
64  
65          this.fieldName = fieldName;
66          this.xpath = adjustXpath(fieldName, xpath);
67          if (format == null || format.length() == 0)
68          {
69              this.format = null;
70          }
71          else
72          {
73              this.format = format;
74          }
75          this.xmlVariable = xmlVariable;
76          this.valueVariable = valueVariable;
77          this.valueAsStringVariable = valueVariable + AS_STRING_SUFFIX;
78          this.valueReaderVariable = valueVariable + READER_SUFFIX;
79      }
80  
81  
82      private String adjustXpath(String defaultValue, String xpath)
83      {
84          String effectiveXpath = xpath;
85          if (effectiveXpath == null || effectiveXpath.length() == 0)
86          {
87              effectiveXpath = defaultValue;
88              if (isPrimitive() || isBasicType() || isEnum())
89              {
90                  effectiveXpath += "/text()";
91              }
92          }
93          return effectiveXpath;
94      }

97      @Override
98      public String toString()
99      {
100         StringBuilder builder = new StringBuilder().append(fieldType.getParameterizedQualifiedSourceName()).append(" ")
101                 .append(fieldName).append(", xpath=\"").append(xpath).append("\"");
102         if (format != null)
103         {
104             builder.append(", format=\"").append(format).append("\"");
105         }
106         return builder.toString();
107     }
